Using Edje Perspective

Edje perspective is a graphical tool that makes 2D objects appear as if they have a 3D appearance. You can use Edje perspective in all Edje objects by setting the perspective to an individual Edje object or the entire canvas. When a perspective is set to the canvas, it affects globally all objects that have no specific perspective already defined.

Creating an Example Application

The perspective example application shows control buttons at the bottom of the screen, and an Edje object above them. You can use the buttons to apply different perspectives to the object: you can move the Edje object (change its perspective point) and change the focal length of the camera (the distance between the object and the camera lens).

To create the example application UI:

  1. In the EDC file, define a simple Edje object that consists of a text and rectangle part.
    Note
    To apply a perspective to an object, it must have the map.perspective_on property set to 1 in the part description.

    The object has 4 states: left-top (default), right-top (right), left-bottom (bottom), and right-bottom (bottomright).

    In addition, create programs that change the object state based on various signals. Later in this topic, one of the control buttons is implemented to emit those signals to make the object move.

    group
{
   name: "example/group";
   min: 480 320;
   parts
   {
      part
      {
         name: "bg";
         type: RECT;
         mouse_events: 1;
         description
         {
            state: "default" 0.0;
         }
      }
      part
      {
         name: "rectangle";
         type: RECT;
         mouse_events: 0;
         description
         {
            state: "default" 0.0;
            color: 255 0 0 128;
            rel1
            {
               offset: -5 -5;
               to: "title";
            }
            rel2
            {
               offset: 4 4;
               to: "title";
            }
            map
            {
               on: 1;
               perspective_on: 1;
               rotation
               {
                  x: 45;
                  y: 15;
               }
            }
         }
      }
      part
      {
         name: "title";
         type: TEXT;
         mouse_events: 0;
         description
         {
            state: "default" 0.0;
            color: 200 200 200 255;
            align: 0.0 0.5;
            rel1.relative: 0.2 0.2;
            rel2.relative: 0.2 0.2;
            text
            {
               text: "Perspective example";
               font: "Sans";
               size: 16;
               min: 1 1;
            }
            map
            {
               on: 1;
               perspective_on: 1;
               rotation
               {
                  x: 45;
                  y: 15;
               }
            }
         }
         description
         {
            state: "right" 0.0;
            inherit: "default" 0.0;
            rel1.relative: 0.5 0.2;
            rel2.relative: 0.5 0.2;
         }
         description
         {
            state: "bottom" 0.0;
            inherit: "default" 0.0;
            rel1.relative: 0.2 0.8;
            rel2.relative: 0.2 0.8;
         }
         description
         {
            state: "bottomright" 0.0;
            inherit: "default" 0.0;
            rel1.relative: 0.5 0.8;
            rel2.relative: 0.5 0.8;
         }
      }
   }
   programs
   {
      program
      {
         name: "move,right";
         signal: "move,1,0";
         action: STATE_SET "right" 0.0;
         transition: SINUSOIDAL 1.0;
         target: "title";
         after: "animation,end";
      }
      program
      {
         name: "move,bottom";
         signal: "move,0,1";
         action: STATE_SET "bottom" 0.0;
         transition: SINUSOIDAL 1.0;
         target: "title";
         after: "animation,end";
      }
      program
      {
         name: "move,bottomright";
         signal: "move,1,1";
         action: STATE_SET "bottomright" 0.0;
         transition: SINUSOIDAL 1.0;
         target: "title";
         after: "animation,end";
      }
      program
      {
         name: "move,default";
         signal: "move,0,0";
         action: STATE_SET "default" 0.0;
         transition: SINUSOIDAL 1.0;
         target: "title";
         after: "animation,end";
      }
      program
      {
         name: "animation,end";
         action: SIGNAL_EMIT "animation,end" "";
      }
   }
}
  2. In the C code, create the application UI components:
    • Create a basic window component that is loaded and created during the application start-up: 
      /* Window */
ad->win = elm_win_util_standard_add(PACKAGE, PACKAGE);
elm_win_conformant_set(ad->win, EINA_TRUE);
elm_win_autodel_set(ad->win, EINA_TRUE);

if (elm_win_wm_rotation_supported_get(ad->win)) {
    int rots[4] =
    {
        0, 90, 180, 270
    };
    elm_win_wm_rotation_available_rotations_set(ad->win, (const int *)(&rots), 4);
}
    • Add a conformant and set the main layout of the application.

      The main layout is described in the EDC part and consists of 2 swallows. The first swallow has the Edje object loaded into it. The second contains a box of buttons to manipulate the perspective of the Edje object. The EDC part of the layout is not covered in this topic.

      /* Conformant */
ad->conform = elm_conformant_add(ad->win);
elm_win_indicator_mode_set(ad->win, ELM_WIN_INDICATOR_SHOW);
elm_win_indicator_opacity_set(ad->win, ELM_WIN_INDICATOR_OPAQUE);
evas_object_size_hint_weight_set(ad->conform, EVAS_HINT_EXPAND, EVAS_HINT_EXPAND);
elm_win_resize_object_add(ad->win, ad->conform);
evas_object_show(ad->conform);

/* Base layout */
app_get_resource(EDJ_FILE, edj_path, (int)PATH_MAX);
ad->layout = elm_layout_add(ad->win);
elm_layout_file_set(ad->layout, edj_path, GRP_MAIN);
evas_object_size_hint_weight_set(ad->layout, EVAS_HINT_EXPAND, EVAS_HINT_EXPAND);
evas_object_size_hint_weight_set(ad->layout, EVAS_HINT_FILL, EVAS_HINT_FILL);
eext_object_event_callback_add(ad->layout, EEXT_CALLBACK_BACK, layout_back_cb, ad);
elm_object_content_set(ad->conform, ad->layout);
    • Create the Edje object:
      • Create the Edje object on Evas using the evas_object_evas_get() function. This function is very useful because you must operate on the canvas, but have only an object pointer to a window object.
      • Instantiate a new Edje object by creating a new Edje smart object and returning its Evas_Object handle with the edje_object_add() function.
      • An Edje object is useless without a source file set to it, so use the edje_object_file_set() function to define the EDC file containing the related group for the object. With the elm_object_part_content_set() function, you can set the object into the main layout's swallow part.
      Evas *evas;
evas = evas_object_evas_get(ad->win);
ad->edje_obj = edje_object_add(evas);
edje_object_file_set(ad->edje_obj, edj_path, "example/group");
evas_object_move(ad->edje_obj, 0, 0);
evas_object_show(ad->edje_obj);
elm_object_part_content_set(ad->layout, "swallow", ad->edje_obj);

    • To apply a perspective to the Edje object, create a perspective object.

      Use the edje_perspective_set() function to define the parameters of the perspective transformation:

      • The second and third parameters define the X and Y axis points that specify the infinite distance point in the 3D conversion, where all lines converge.
      • The fourth parameter specifies the Z axis value at which there is a 1:1 mapping between the spatial and screen coordinates. The points on the Z value do not have their X and Y values modified in the transformation. The points further away (with a higher Z value) shrink into the distance, and those that are closer expand and become bigger.
      • The last parameter determines the focal length of the camera, allowing for depth control. The value is the distance between the camera lens plane and the Z axis value, and it must be greater than 0.

      To set a global perspective (that applies to the entire canvas), use the edje_perspective_global_set() function. To set a perspective that is applied only to a specific Edje object, use the edje_object_perspective_set() function.

      ad->ps = edje_perspective_new(evas);
edje_perspective_set(ad->ps, 160, 320, 0, ad->focal);
edje_perspective_global_set(ad->ps, EINA_TRUE);

    • Create the control buttons to manipulate the perspective.

      The buttons are packed into a box, whose style is defined in the swallow.buttons layout part in the EDC file (not covered in this topic).

      Add the box, pack the move, more, and less buttons into it, and register callbacks for the button clicks:

      Evas_Object *box;
Evas_Object *button;
/* Adding a box */
box = elm_box_add(ad->layout);
evas_object_size_hint_weight_set(box, EVAS_HINT_EXPAND, 0.0);
evas_object_size_hint_align_set(box, EVAS_HINT_FILL, 0.0);
elm_box_horizontal_set(box, EINA_TRUE);
evas_object_show(box);
elm_object_part_content_set(ad->layout, "swallow.buttons", box);

/* Adding the buttons */
button = elm_button_add(ad->layout);
evas_object_size_hint_align_set(button, EVAS_HINT_FILL, EVAS_HINT_FILL);
evas_object_size_hint_weight_set(button, EVAS_HINT_EXPAND, EVAS_HINT_EXPAND);
elm_object_text_set(button, "move");
evas_object_show(button);
elm_box_pack_end(box, button);
evas_object_smart_callback_add(button, "clicked", _on_btn_clicked, ad);

button = elm_button_add(ad->layout);
evas_object_size_hint_align_set(button, EVAS_HINT_FILL, EVAS_HINT_FILL);
evas_object_size_hint_weight_set(button, EVAS_HINT_EXPAND, EVAS_HINT_EXPAND);
elm_object_text_set(button, "more");
evas_object_show(button);
elm_box_pack_end(box, button);
evas_object_smart_callback_add(button, "clicked", _on_btn_more_clicked, ad);

button = elm_button_add(ad->layout);
evas_object_size_hint_align_set(button, EVAS_HINT_FILL, EVAS_HINT_FILL);
evas_object_size_hint_weight_set(button, EVAS_HINT_EXPAND, EVAS_HINT_EXPAND);
elm_object_text_set(button, "less");
evas_object_show(button);
elm_box_pack_end(box, button);
evas_object_smart_callback_add(button, "clicked", _on_btn_less_clicked, ad);

    • Show the main window:

      evas_object_show(ad->win);

Managing the Perspective

To manage the perspective in the example application, use the move, more, and less buttons:

  1. Use the move button click callback to move the Edje object around the screen. Since the perspective point remains the same, the changing position causes the object to change its appearance.

    In the callback, call the part_move() function to send a signal to the Edje part in the EDC file with the edje_object_signal_emit() function. The signal is handled in the applicable program block, causing the Edje part state to change and consequently move the object to a new position on the screen.

    static void
_part_move(appdata_s *ad, int dx, int dy)
{
    char emission[64];

    dlog_print(DLOG_ERROR, "AAAA", "x,y = %d, %d", dx, dy);
    snprintf(emission, sizeof(emission), "move,%d,%d", dx, dy);
    edje_object_signal_emit(ad->edje_obj, emission, "");
}

static void
_on_btn_clicked(void *data, Evas_Object *obj, void *event_info)
{
    appdata_s *ad = data;
    static int i = 0;
    switch (i) {
    case 0:
        _part_move(ad, 1, 0);
        break;
    case 1:
        _part_move(ad, 1, 1);
        break;
    case 2:
        _part_move(ad, 0, 1);
        break;
    case 3:
        _part_move(ad, 0, 0);
        break;
    }
    i++;
    i = i%4;
}

  2. Use the more and less button click callbacks to increase or decrease the focal length of the camera.

    In the callbacks, change the focal length value, reset the parameters of the perspective transformation with the edje_perspective_set() function, and apply the new perspective globally.

    /* Increase the focal length with the more button */
static void
_on_btn_more_clicked(void *data, Evas_Object *obj, void *event_info)
{
    appdata_s *ad = data;
    ad->focal += 5;
    edje_perspective_set(ad->ps, 160, 320, 0, ad->focal);
    edje_perspective_global_set(ad->ps, EINA_TRUE);
}

/* Decrease the focal length with the less button */
static void
_on_btn_less_clicked(void *data, Evas_Object *obj, void *event_info)
{
    appdata_s *ad = data;
    ad->focal -= 5;
    edje_perspective_set(ad->ps, 160, 320, 0, ad->focal);
    edje_perspective_global_set(ad->ps, EINA_TRUE);
}
